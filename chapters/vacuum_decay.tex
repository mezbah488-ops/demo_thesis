\chapter{False Vacuum Decay and Bubble Nucleation}
\label{ch:vacuum-decay}

Having established that first-order phase transitions proceed via the
nucleation of bubbles of the true vacuum within the false vacuum, we
now develop the quantitative framework for computing the nucleation
rate. The central object is the bounce solution, a saddle-point of the
Euclidean action that dominates the path integral for the false vacuum
decay rate. We begin with quantum mechanics, where the essential
physics is transparent, and then generalize to quantum field theory.

\section{Decay Rate in Quantum Mechanics}

Consider a quantum mechanical system with a metastable ground state
localized at $x_0$. If the energy eigenvalue is complex,
\begin{equation}
    E_0 = \text{Re}\,E_0 + i\,\text{Im}\,E_0,
\end{equation}
the norm of the wave function decays as
\begin{align}
    |\psi_0(t)|^2 &= \left|e^{-iHt}\psi_0(0)\right|^2 \notag\\
    &= e^{2\,\text{Im}\,E_0\,t}|\psi_0(0)|^2 \notag\\
    &= e^{-\Gamma t}|\psi_0(0)|^2,
\end{align}
where $\Gamma = -2\,\text{Im}\,E_0$ is the decay rate. To compute
$E_0$, we use the path integral representation of the transition
amplitude,
\begin{equation}
    \langle x_b|e^{-i(t_b-t_a)H}|x_a\rangle
    = \int_{x(t_a)=x_a}^{x(t_b)=x_b}\mathcal{D}x\,e^{iS(x)},
    \label{eq:tr_amp}
\end{equation}
with
\begin{equation}
    S(x) = \int_{t_a}^{t_b}dt\left[
    \frac{1}{2}\left(\frac{dx}{dt}\right)^2 - V(x)\right].
\end{equation}
We proceed in three steps. First, set $|x_b\rangle = |x_a\rangle =
|x_0\rangle$. Second, Wick rotate $t = -i\tau$ with $\tau$ real.
Third, set $\tau_a = -\hat\tau/2$ and $\tau_b = \hat\tau/2$.

The left-hand side of \eqref{eq:tr_amp} becomes
\begin{equation}
    \langle x_0|e^{-\hat H\hat\tau}|x_0\rangle
    = \sum_n e^{-E_n\hat\tau}|\langle x_0|\psi_n\rangle|^2
    \;\overset{\hat\tau\to\infty}{=}\;
    e^{-E_0\hat\tau}|\langle x_0|\psi_0\rangle|^2,
\end{equation}
where all excited states are exponentially suppressed. The ground state
energy is therefore
\begin{equation}
    E_0 = -\lim_{\hat\tau\to\infty}\frac{1}{\hat\tau}
    \log\langle x_0|e^{-\hat H\hat\tau}|x_0\rangle.
\end{equation}
The right-hand side of \eqref{eq:tr_amp} after Wick rotation becomes
\begin{equation}
    \int\mathcal{D}x\,e^{-S_E},
    \label{eq:Euclidean_integral}
\end{equation}
where the Euclidean action is
\begin{equation}
    S_E = \int_{-\hat\tau/2}^{\hat\tau/2}d\tau\left[
    \frac{1}{2}\left(\frac{dx}{d\tau}\right)^2 + V(x)\right].
    \label{eq:Euclidean_action}
\end{equation}
We evaluate \eqref{eq:Euclidean_integral} by expanding around a
stationary point $\bar x(\tau)$ of $S_E$. The Euler-Lagrange equation
gives
\begin{equation}
    \frac{d^2 x(\tau)}{d\tau^2} = V'(x),
    \label{eq:eom}
\end{equation}
with boundary condition $x(-\hat\tau/2) = x(\hat\tau/2) = x_0$.
Crucially, \eqref{eq:eom} describes a classical particle moving in the
\emph{inverted} potential $-V(x)$.

% [FIGURE: original potential V(x) with false vacuum at x_0 and true
%  vacuum at x_1, and inverted potential -V(x) showing the bounce
%  trajectory]

Setting $x(\tau) = \bar x(\tau) + z(\tau)$ and expanding $S_E$,
\begin{align}
    S_E(x) &= S_E(\bar x)
    - \int d\tau\left(\frac{d^2\bar x}{d\tau^2} - V'(\bar x)\right)z
    \notag\\
    &\quad+ \frac{1}{2}\int d\tau\,z(\tau)
    \left[-\frac{d^2}{d\tau^2} + V''(\bar x)\right]z(\tau)
    + \mathcal{O}(z^3),
\end{align}
where integration by parts was used and boundary terms dropped. The
linear term vanishes by the equation of motion, leaving
\begin{equation}
    \int\mathcal{D}x\,e^{-S_E}
    = e^{-S_E(\bar x)}\int\mathcal{D}z\,
    \exp\!\left(-\frac{1}{2}\int d\tau\,z
    \left[-\frac{d^2}{d\tau^2}+V''(\bar x)\right]z\right).
\end{equation}
The Gaussian integral over $z$ gives a functional determinant,
\begin{equation}
    = N\left[\det\!\left(-\frac{d^2}{d\tau^2}
    + V''(\bar x)\right)\right]^{-1/2}
    \;\overset{\hat\tau\to\infty}{=}\;
    \left(\frac{\omega}{\pi}\right)^{1/2}e^{-\omega\hat\tau/2},
\end{equation}
where $\omega = \sqrt{V''(x_0)}$. For the trivial stationary path
$\bar x(\tau) = x_0$, we have $S_E = 0$ (choosing $V(x_0) = 0$),
and the transition amplitude gives
\begin{equation}
    \langle x_0|e^{-H\hat\tau}|x_0\rangle
    = \left(\frac{\omega}{\pi}\right)^{1/2}e^{-\omega\hat\tau/2},
\end{equation}
from which we read off the familiar result $E_0 = \omega/2$.

\section{The Bounce Solution}

The non-trivial stationary path is the bounce: a solution to
\eqref{eq:eom} in the inverted potential $-V(x)$ that starts at $x_0$
at $\tau = -\infty$, rolls to the turning point $x_*$, and returns to
$x_0$ at $\tau = +\infty$.

% [FIGURE: bounce trajectory bar{x}(tau) as a function of tau,
%  showing the localized instanton profile]

Let $\bar x(\tau)$ be a bounce solution with Euclidean action
$S_E(\bar x) = B$. Its contribution to the path integral, together
with the correction factor
\begin{equation}
    K = \left[\frac{\det'(-\partial_\tau^2 + V''(\bar x))}
    {\det(-\partial_\tau^2 + \omega^2)}\right]^{-1/2},
\end{equation}
where the prime on $\det'$ denotes exclusion of the zero eigenvalue,
is $e^{-B}K$.

\section{The Dilute Gas Approximation}

For $n$ widely separated bounces with centers
$\tau_1 \ll \tau_2 \ll \cdots \ll \tau_n$, the contributions
multiply. Integrating over the positions of all centers gives a factor
\begin{equation}
    \int_{-\hat\tau/2}^{\hat\tau/2}d\tau_1
    \int_{\tau_1}^{\hat\tau/2}d\tau_2\cdots
    \int_{\tau_{n-1}}^{\hat\tau/2}d\tau_n
    = \frac{\hat\tau^n}{n!}.
\end{equation}
Summing over all $n$,
\begin{align}
    \int\mathcal{D}x\,e^{-S_E}
    &= \left(\frac{\omega}{\pi}\right)^{1/2}e^{-\omega\hat\tau/2}
    \sum_{n=0}^{\infty}\frac{\hat\tau^n}{n!}e^{-Bn}K^n \notag\\
    &= \left(\frac{\omega}{\pi}\right)^{1/2}
    e^{-\hat\tau(\omega/2 - Ke^{-B})},
\end{align}
from which we read off the ground state energy including the
instanton correction,
\begin{equation}
    \boxed{E_0 = \frac{\omega}{2} - Ke^{-B}.}
\end{equation}
The dilute gas approximation is justified when $B\gg 1$: the series
$\sum_n \hat\tau^n e^{-Bn}K^n/n!$ is dominated by terms with
$n\lesssim \hat\tau Ke^{-B}$, so the bounce density $n/\hat\tau\sim
Ke^{-B}$ is exponentially suppressed.

There are two subtleties in the computation of $K$. First, the
operator $(-\partial_\tau^2 + V''(\bar x))$ has a zero eigenvalue,
corresponding to translations of the bounce center. This zero mode
must be excluded from the determinant and treated separately,
contributing a factor $(B/2\pi)^{1/2}$. Second, the operator has a
negative eigenvalue, whose treatment requires analytic continuation.
Following Callan and Coleman, the result is
\begin{equation}
    \text{Im}\,K = \frac{1}{2}\left(\frac{B}{2\pi}\right)^{1/2}
    \left[\frac{\det'(-\partial_\tau^2+V''(\bar x))}
    {\det(-\partial_\tau^2+\omega^2)}\right]^{-1/2}e^{-B},
\end{equation}
giving a decay rate per unit time
\begin{equation}
    \Gamma = \left(\frac{B}{2\pi}\right)^{1/2}
    \left[\frac{\det'(-\partial_\tau^2+V''(\bar x))}
    {\det(-\partial_\tau^2+\omega^2)}\right]^{-1/2}e^{-B}.
\end{equation}

\section{Decay Rate in Quantum Field Theory}

The generalization to QFT is straightforward. The transition amplitude
\begin{equation}
    \langle\phi_b|e^{-H(\tau_b-\tau_a)}|\phi_a\rangle
    = \int\mathcal{D}\phi\,e^{-S_E[\phi]}
\end{equation}
is evaluated by the same saddle-point procedure. The decay rate per
unit time and volume becomes
\begin{equation}
    \Gamma = \frac{B^2}{4\pi^2}
    \left[\frac{\det'(-\partial_\tau^2-\nabla^2+V''(\bar\phi))}
    {\det(-\partial_\tau^2-\nabla^2+\omega^2)}\right]^{-1/2}e^{-B},
\end{equation}
where the bounce action is
\begin{equation}
    B = S_E[\bar\phi] = \int d\tau\,d^3x\left[
    \frac{1}{2}(\partial_\tau\bar\phi)^2
    + \frac{1}{2}(\boldsymbol\nabla\bar\phi)^2
    + V(\bar\phi)\right].
\end{equation}
The functional determinant is difficult to compute but contributes
only a polynomial prefactor; the exponential suppression $e^{-B}$
dominates and is the quantity of primary interest.

The bounce solution $\bar\phi$ satisfies the Euclidean equation of
motion
\begin{equation}
    (\partial_\tau^2 + \nabla^2)\bar\phi - V'(\bar\phi) = 0,
    \label{eq:bounce_eom}
\end{equation}
with boundary conditions
\begin{equation}
    \lim_{\tau\to\pm\infty}\bar\phi(\tau,\mathbf{x}) = \phi_0,
    \qquad
    \lim_{|\mathbf{x}|\to\infty}\bar\phi(\tau,\mathbf{x}) = \phi_0,
\end{equation}
the second of which ensures the action is finite. By a theorem of
Coleman, Glashow, and Martin (1978), the $O(4)$-symmetric solution
has the lowest action. Writing $\bar\phi(\tau,\mathbf{x}) = \phi(\rho)$
with $\rho = \sqrt{\tau^2+|\mathbf{x}|^2}$, \eqref{eq:bounce_eom}
reduces to
\begin{equation}
    \frac{d^2\bar\phi}{d\rho^2} + \frac{3}{\rho}\frac{d\bar\phi}{d\rho}
    = V'(\bar\phi),
    \label{eq:spherical_eom}
\end{equation}
with boundary conditions $\bar\phi'(0) = 0$ and
$\bar\phi(\rho)\to\phi_0$ as $\rho\to\infty$. This is analogous to a
classical particle evolving in fictitious time $\rho$, subject to a
time-dependent damping force $3\rho^{-1}d/d\rho$, moving in the
inverted potential $-V(\bar\phi)$.

%  near rho=0 and phi_0 at rho=infinity]

\section{The Overshoot-Undershoot Algorithm}

Equation \eqref{eq:spherical_eom} is solved numerically by the
overshoot-undershoot algorithm. Starting from an initial value
$\bar\phi(0) = \phi_i$ with $\phi_0\leq\phi_i\leq\phi_1$ and zero
initial velocity $\bar\phi'(0) = 0$, one evolves to large $\rho$. Two
outcomes are possible. If $\phi_i$ is too large, the field gains too
much kinetic energy and overshoots, passing below $\phi_0$. If $\phi_i$
is too small, the field undershoots and never reaches $\phi_0$. The
correct bounce solution lies between these two cases and is found by
binary search, halving the step size at each iteration until the
algorithm converges rapidly.

The two limiting cases are the thin-wall and thick-wall approximations,
corresponding respectively to a potential barrier that is narrow
compared to the bubble radius and one that is wide.

\section{Bubble Nucleation and Dynamics}

The bounce solution has a dual role: it dominates the functional
integral for the false vacuum decay rate, and it describes the field
configuration immediately after tunneling. The initial bubble is given
by
\begin{align}
    \phi_{\text{bubble}}(t=0,\mathbf{x})
    &= \bar\phi(\tau=0,\mathbf{x}), \\
    \frac{\partial}{\partial t}\phi_{\text{bubble}}(t,\mathbf{x})
    \big|_{t=0} &= 0.
\end{align}
After nucleation, the bubble evolves according to the Minkowski
equation of motion
\begin{equation}
    -\frac{\partial^2}{\partial t^2}\phi_{\text{bubble}}
    + \nabla^2\phi_{\text{bubble}} = V'(\phi_{\text{bubble}}),
    \label{eq:bubble_eom}
\end{equation}
whose solution is obtained by analytic continuation from Euclidean
space,
\begin{equation}
    \phi_{\text{bubble}}(t,\mathbf{x})
    = \bar\phi\!\left(\sqrt{|\mathbf{x}|^2 - t^2}\right).
\end{equation}
The bubble wall is the transition region between the true and false
vacuum phases. Its position satisfies
$|\mathbf{x}| = \sqrt{t^2 + C^2}$, where $C$ is a constant, and the
wall velocity approaches the speed of light at late times.

% [FIGURE: bubble wall expanding in spacetime, showing the Lorentz
%  contraction of the wall and the true vacuum interior]

The functional determinant prefactor is of order
$\mathcal{O}(R^4, \bar\phi(0)^4, V''(\bar\phi))$, where $R$ is the
bubble radius at nucleation. Since it contributes only polynomially
compared to the exponential $e^{-B}$, the dominant quantity
determining the nucleation rate is the bounce action $B$.

\section{Finite Temperature and the \texorpdfstring{$O(3)$}{O(3)}
Bounce}

At finite temperature the Euclidean time direction is compactified to
a circle of circumference $\beta = T^{-1}$. When the temperature is
high, $T\gg 1/R$ where $R$ is the bubble radius, the $O(4)$-symmetric
instanton transitions to an $O(3)$-symmetric instanton. This is
because the thermal fluctuations dominate quantum tunneling, and the
bounce becomes time-independent.

% [FIGURE: O(4) symmetric bubble at low T versus O(3) symmetric
%  bubble at high T, with compactified Euclidean time]

Setting $\partial_\tau\bar\phi = 0$, the Euclidean action reduces to
\begin{equation}
    S_E = \frac{1}{T}\underbrace{\int d^3x\left[
    \frac{1}{2}(\boldsymbol\nabla\bar\phi)^2
    + V(\bar\phi)\right]}_{S_3},
\end{equation}
and the $O(3)$ equation of motion is
\begin{equation}
    \frac{d^2\bar\phi}{d\rho^2}
    + \frac{2}{\rho}\frac{d\bar\phi}{d\rho} = V'(\bar\phi),
    \qquad \rho = |\mathbf{x}|.
\end{equation}
The two cases are summarized as
\begin{equation}
    \frac{d^2\phi}{d\rho^2} + \frac{c_i}{\rho}\frac{d\phi}{d\rho}
    = V'(\phi),
    \qquad c_i =
    \begin{cases}
        3 & O(4)\text{ symmetry (}T=0\text{)}, \\
        2 & O(3)\text{ symmetry (high }T\text{)},
    \end{cases}
\end{equation}
with corresponding bounce actions
\begin{equation}
    B = \begin{cases}
        S_E[\bar\phi] & O(4)\text{ symmetry}, \\
        S_3/T & O(3)\text{ symmetry}.
    \end{cases}
\end{equation}
At finite temperature the effective potential $V(\bar\phi)$ is
replaced by the finite-temperature effective potential
$V_{\text{eff}}^\beta(\bar\phi)$ computed in Part~\ref{part:finite-temp},
and the nucleation rate becomes
\begin{equation}
    \Gamma \sim e^{-S_3/T}.
\end{equation}
The nucleation temperature $T_n$ is defined as the temperature at
which one bubble per Hubble volume per Hubble time nucleates, and the
percolation temperature $T_p$ is the temperature at which the phase
transition completes. These parameters, together with the latent heat
$\alpha$ and the transition rate $\tilde\beta$, determine the
gravitational wave spectrum produced during the transition.
