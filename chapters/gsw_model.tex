\chapter{The Glashow-Salam-Weinberg Model}
\label{ch:gsw-model}

The Glashow-Salam-Weinberg (GSW) model is the quantum field theory of
electroweak interactions, unifying the electromagnetic and weak forces
into a single gauge theory with gauge group $\mathrm{SU}(2)_L \times
	\mathrm{U}(1)_Y$. It is one of the cornerstones of the Standard Model
and provides the first physical example of spontaneous symmetry breaking
in a gauge theory, the Higgs mechanism, which we studied abstractly in
the previous chapter. Here we apply the effective potential formalism to
this model, computing the field-dependent gauge boson masses and the
resulting one-loop correction to the effective potential.

\section{The Electroweak Lagrangian}

The gauge fields of the theory are the $\mathrm{SU}(2)_L$ triplet
$W_\mu^i$ ($i=1,2,3$) and the $\mathrm{U}(1)_Y$ singlet $V_\mu$, with
corresponding field strengths $W_{\mu\nu}^i$ and $V_{\mu\nu}$. The
matter content includes left-handed fermion doublets and right-handed
singlets, but for the purposes of computing the effective potential it
is the scalar sector that is relevant. The Higgs field $\varphi$ is a
complex doublet under $\mathrm{SU}(2)_L$ with hypercharge $Y=1$. The
Lagrangian is
\begin{equation}
	\mathcal{L} = -\frac{1}{4}W_{\mu\nu}^i W^{i\,\mu\nu}
	- \frac{1}{4}V_{\mu\nu}V^{\mu\nu}
	+ (D_\mu\varphi)^\dagger(D^\mu\varphi)
	- V(\varphi),
\end{equation}
where the covariant derivative is
\begin{equation}
	D_\mu = \partial_\mu - ig\frac{\sigma^i}{2}W_\mu^i
	- ig'\frac{Y}{2}V_\mu,
\end{equation}
with $g$ and $g'$ the $\mathrm{SU}(2)_L$ and $\mathrm{U}(1)_Y$ coupling
constants respectively, and $\sigma^i$ the Pauli matrices. The classical
potential $V(\varphi) = -\mu^2\varphi^\dagger\varphi +
	\lambda(\varphi^\dagger\varphi)^2$ has a minimum at a nonzero field
value when $\mu^2 > 0$, triggering spontaneous symmetry breaking.

\section{Spontaneous Symmetry Breaking and Mass Generation}

The Higgs field acquires a vacuum expectation value
\begin{equation}
	\langle\varphi\rangle = \begin{pmatrix}0\\a\end{pmatrix},
\end{equation}
which breaks $\mathrm{SU}(2)_L\times\mathrm{U}(1)_Y$ down to the
electromagnetic $\mathrm{U}(1)_{\text{em}}$. The gauge boson masses are
read off from the kinetic term evaluated at the VEV. Since the VEV is
constant, the $\partial_\mu$ term vanishes and
\begin{equation}
	D_\mu\langle\varphi\rangle
	= \left(-ig\frac{\sigma^i}{2}W_\mu^i
	- ig'\frac{Y}{2}V_\mu\right)\begin{pmatrix}0\\a\end{pmatrix}.
\end{equation}
Expanding the $\mathrm{SU}(2)$ part using
\begin{equation}
	\sigma^i W_\mu^i =
	\begin{pmatrix}W_\mu^3          & W_\mu^1-iW_\mu^2 \\
	               W_\mu^1+iW_\mu^2 & -W_\mu^3\end{pmatrix},
\end{equation}
acting on the VEV gives
\begin{equation}
	\frac{\sigma^i}{2}W_\mu^i\begin{pmatrix}0\\a\end{pmatrix}
	= \frac{a}{2}\begin{pmatrix}W_\mu^1-iW_\mu^2\\-W_\mu^3\end{pmatrix}.
\end{equation}
For the $\mathrm{U}(1)$ part, with hypercharge $Y=1$,
\begin{equation}
	\frac{Y}{2}V_\mu\begin{pmatrix}0\\a\end{pmatrix}
	= \frac{a}{2}\begin{pmatrix}0\\V_\mu\end{pmatrix}.
\end{equation}
Combining,
\begin{equation}
	D_\mu\langle\varphi\rangle
	= -ig\cdot\frac{a}{2}
	\begin{pmatrix}W_\mu^1-iW_\mu^2\\-W_\mu^3\end{pmatrix}
	- ig'\cdot\frac{a}{2}\begin{pmatrix}0\\V_\mu\end{pmatrix}
	= \frac{ia}{2}\begin{pmatrix}gW_\mu^1-igW_\mu^2 \\
		-gW_\mu^3+g'V_\mu\end{pmatrix}.
\end{equation}
Taking the conjugate,
\begin{equation}
	\overline{D_\mu\langle\phi\rangle}
	= -\frac{ia}{2}\left(gW_\mu^1+igW_\mu^2,\;
	-gW_\mu^3+g'V_\mu\right),
\end{equation}
so that
\begin{equation}
	\overline{D_\mu\langle\phi\rangle}\,D^\mu\langle\phi\rangle
	= \frac{a^2}{4}\left(gW_\mu^1+igW_\mu^2,\;
	-gW_\mu^3+g'V_\mu\right)
	\begin{pmatrix}gW^{1\mu}-igW^{2\mu} \\
		-gW^{3\mu}+g'V^\mu\end{pmatrix}.
\end{equation}
Expanding the matrix product,
\begin{equation}
	= \frac{a^2}{4}\left[g^2(W_\mu^{1\,2}+W_\mu^{2\,2})
		+ g^2W_\mu^{3\,2} + g'^2V_\mu^2
		- 2gg'W_\mu^3 V_\mu\right].
\end{equation}
The fields $W_\mu^3$ and $V_\mu$ mix, and to identify the physical mass
eigenstates we introduce the Weinberg rotation,
\begin{align}
	A_\mu & = \sin\theta_W\,W_\mu^3 + \cos\theta_W\,V_\mu, \\
	Z_\mu & = \cos\theta_W\,W_\mu^3 - \sin\theta_W\,V_\mu,
\end{align}
where $\sin\theta_W = g'/\sqrt{g^2+g'^2}$ and
$\cos\theta_W = g/\sqrt{g^2+g'^2}$. Inverting,
\begin{align}
	W_\mu^3 & = \sin\theta_W\,A_\mu + \cos\theta_W\,Z_\mu, \\
	V_\mu   & = \cos\theta_W\,A_\mu - \sin\theta_W\,Z_\mu.
\end{align}
Substituting into each term,
\begin{multline}
	g^2W_\mu^{3\,2} = g^2\sin^2\theta_W A_\mu^2
	+ g^2\cos^2\theta_W Z_\mu^2
	+ 2g^2\sin\theta_W\cos\theta_W A_\mu Z_\mu, \\
	g'^2V_\mu^2 = g'^2\cos^2\theta_W A_\mu^2
	+ g'^2\sin^2\theta_W Z_\mu^2
	- 2g'^2\sin\theta_W\cos\theta_W A_\mu Z_\mu, \\
	-2gg'W_\mu^3 V_\mu
	= -2gg'(\sin\theta_W A_\mu+\cos\theta_W Z_\mu)
	(\cos\theta_W A_\mu-\sin\theta_W Z_\mu) \\
	= -2gg'\sin\theta_W\cos\theta_W A_\mu^2
	+ 2gg'\sin^2\theta_W A_\mu Z_\mu
	- 2gg'\cos^2\theta_W Z_\mu A^\mu
	+ 2gg'\cos\theta_W\sin\theta_W Z_\mu^2.
\end{multline}
Collecting by field bilinear,
\begin{align}
	g^2W_\mu^{3\,2} + g'^2V_\mu^2 - 2gg'W_\mu^3 V^\mu
	 & = A_\mu^2\left(g\sin\theta_W-g'\cos\theta_W\right)^2 \notag \\
	 & + Z_\mu^2\left(g\cos\theta_W+g'\sin\theta_W\right)^2 \notag \\
	 & + 2\sin\theta_W\cos\theta_W A_\mu Z^\mu
	\left(g^2-g'^2+gg'\left(\frac{\sin\theta_W}{\cos\theta_W}
	-\frac{\cos\theta_W}{\sin\theta_W}\right)\right).
\end{align}
The coefficient of $A_\mu^2$ vanishes identically,
\begin{equation}
	g\sin\theta_W - g'\cos\theta_W
	= \frac{gg'}{\sqrt{g^2+g'^2}}
	- \frac{g'g}{\sqrt{g^2+g'^2}} = 0,
\end{equation}
and the $A_\mu Z^\mu$ cross term also vanishes. The coefficient of
$Z_\mu^2$ evaluates to
\begin{equation}
	\left(g\cos\theta_W+g'\sin\theta_W\right)^2
	= \left(\frac{g^2}{\sqrt{g^2+g'^2}}
	+\frac{g'^2}{\sqrt{g^2+g'^2}}\right)^2
	= g^2+g'^2.
\end{equation}
The photon therefore remains massless, as required by the unbroken
$\mathrm{U}(1)_{\text{em}}$, and the kinetic term becomes
\begin{equation}
	\boxed{\overline{D_\mu\langle\varphi\rangle}\,D^\mu\langle\varphi\rangle
		= \frac{a^2}{4}\left[g^2\left(W_\mu^{1\,2}+W_\mu^{2\,2}\right)
			+ \left(g^2+g'^2\right)Z_\mu^2\right].}
\end{equation}
Reading off the field-dependent masses,
\begin{align}
	m^2(W_{1,2}) & = \frac{1}{4}g^2 a^2,       \\
	m^2(Z)       & = \frac{1}{4}(g^2+g'^2)a^2,
\end{align}
while the photon remains massless. These are the standard results of the
Higgs mechanism in the electroweak theory.

\section{One-Loop Gauge Field Contribution to the Effective Potential}

With the field-dependent gauge boson masses identified, we can compute
their contribution to the one-loop effective potential. By the same logic
as in the scalar case, each massive gauge boson with field-dependent mass
matrix $M^2$ contributes a Coleman-Weinberg type term at one loop. The
gauge field contribution is
\begin{equation}
	V_g^{(1)}(\bar\varphi) = \frac{3}{64\pi^2}
	\,\mathrm{Tr}\!\left(M^4\ln M^2\right)\Big|_{\rho=\bar\varphi},
	\qquad
	M^2_{\alpha\beta} = \lambda_\alpha\lambda_\beta\,
	\mathrm{Tr}\!\left[(t_\alpha\varphi)^\dagger\tau_\beta\varphi\right],
\end{equation}
where the factor of 3 accounts for the three physical polarizations of
each massive gauge boson. Substituting the masses of the two $W$ bosons
and the $Z$ boson,
\begin{align}
	V_g^{(1)}(\bar\varphi)
	 & = \frac{3}{64\pi^2}\left(\frac{1}{16}g^4
	+ \frac{1}{16}g^4
	+ \frac{1}{16}(g^2+g'^2)^2\right)
	\bar\varphi^4\ln\bar\varphi^2 \notag        \\
	 & = \frac{3}{64\pi^2}\cdot\frac{1}{16}
	\left[2g^4+(g^2+g'^2)^2\right]
	\bar\varphi^4\ln\bar\varphi^2,
\end{align}
\begin{equation}
	\boxed{\Rightarrow\quad V_g^{(1)}(\bar\varphi)
		= \frac{3}{1024\pi^2}\left[2g^4+(g^2+g'^2)^2\right]
		\bar\varphi^4\ln\bar\varphi^2.}
\end{equation}
The coefficient is determined entirely by the gauge couplings $g$ and
$g'$, and the structure $\bar\varphi^4\ln\bar\varphi^2$ is the
characteristic signature of the Coleman-Weinberg mechanism in a gauge
theory. In the next chapter we extend this analysis to GUT models, where
the larger gauge group and richer symmetry breaking pattern lead to
qualitatively new features.
